Get-Started

Below are the first steps you can take to access and start using the API.


Requesting 100 People:

Each of the tabs show example code for how to request 100 people in different languages:

console.log( fetch('https://peoplegeneratorapi.live/api/person/100') .then(response => response.json()) .then(data => console.log(data)) .catch(error => console.error(error)) );
import java.net.URI; import java.net.http.HttpClient; import java.net.http.HttpRequest; import java.net.http.HttpResponse; public class Main { public static void main(String[] args) { HttpClient client = HttpClient.newHttpClient(); HttpRequest request = HttpRequest.newBuilder() .uri(URI.create("https://peoplegeneratorapi.live/api/person/100")) .GET() .build(); try { HttpResponse<String>response = client.send(request, HttpResponse.BodyHandlers.ofString()); System.out.println(response.body()); } catch (Exception e) { e.printStackTrace(); } } }
import requests response = requests.get('https://peoplegeneratorapi.live/api/person/100') response = response.json() # Encode the response to utf-8 so that it can be printed encoded_response = str(response).encode('utf-8') print(encoded_response)
